<script>
  let array = [];
  const size = 30;
  const speed = 50;

  function sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }

  function generateArray() {
    array = [];
    for (let i = 0; i < size; i++) {
      array.push(Math.floor(Math.random() * 200) + 20);
    }
    render();
  }

  function render(active = [], swap = []) {
    const container = document.getElementById('bars');
    container.innerHTML = '';

    array.forEach((value, index) => {
      const bar = document.createElement('div');
      bar.classList.add('bar');

      if (active.includes(index)) bar.classList.add('active');
      if (swap.includes(index)) bar.classList.add('swap');

      bar.style.height = value + 'px';
      container.appendChild(bar);
    });
  }

  async function bubbleSort() {
    for (let i = 0; i < array.length; i++) {
      for (let j = 0; j < array.length - i - 1; j++) {
        render([j, j+1]);
        await sleep(speed);

        if (array[j] > array[j + 1]) {
          [array[j], array[j + 1]] = [array[j + 1], array[j]];
          render([], [j, j+1]);
          await sleep(speed);
        }
      }
    }
    render();
  }

  async function insertionSort() {
    for (let i = 1; i < array.length; i++) {
      let key = array[i];
      let j = i - 1;

      while (j >= 0 && array[j] > key) {
        render([j, j+1]);
        await sleep(speed);

        array[j + 1] = array[j];
        render([], [j, j+1]);
        await sleep(speed);

        j--;
      }

      array[j + 1] = key;
    }
    render();
  }

  generateArray();
</script>

  <br>
  <input id="targetInput" type="number" placeholder="Искомое число">

<div class="bars" id="bars"></div>

<script>
  let array = [];
  const size = 30;
  const speed = 80;

  function sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }

  function generateArray() {
    array = [];
    for (let i = 0; i < size; i++) {
      array.push(Math.floor(Math.random() * 100));
    }
    render();
  }

  function render(active = [], checked = [], found = -1) {
    const container = document.getElementById('bars');
    container.innerHTML = '';

    array.forEach((value, index) => {
      const bar = document.createElement('div');
      bar.classList.add('bar');

      if (active.includes(index)) bar.classList.add('active');
      if (checked.includes(index)) bar.classList.add('checked');
      if (index === found) bar.classList.add('found');

      bar.style.height = (value * 2) + 'px';
      bar.textContent = value;

      container.appendChild(bar);
    });
  }

  async function linearSearch() {
    const target = parseInt(document.getElementById('targetInput').value);
    if (isNaN(target)) return alert('Введите число');

    for (let i = 0; i < array.length; i++) {
      render([i]);
      await sleep(speed);

      if (array[i] === target) {
        render([], [], i);
        return;
      }

      render([], [i]);
    }
    alert('Не найдено');
  }

  async function binarySearch() {
    const target = parseInt(document.getElementById('targetInput').value);
    if (isNaN(target)) return alert('Введите число');

    array.sort((a, b) => a - b);
    render();
    await sleep(500);

    let left = 0;
    let right = array.length - 1;

    while (left <= right) {
      let mid = Math.floor((left + right) / 2);

      render([mid]);
      await sleep(speed);

      if (array[mid] === target) {
        render([], [], mid);
        return;
      }

      if (array[mid] < target) {
        for (let i = left; i <= mid; i++) {
          render([], [i]);
          await sleep(10);
        }
        left = mid + 1;
      } else {
        for (let i = mid; i <= right; i++) {
          render([], [i]);
          await sleep(10);
        }
        right = mid - 1;
      }
    }

    alert('Не найдено');
  }

  generateArray();
</script>

</body>
